// ============================================
// Viz2D Sample Code - Shape Reference & Demo
// ============================================
// Copy this code into StartViz.cs to test all shapes
// ============================================

namespace StartViz
{
    public class Viz
    {
        public static void Main()
        {
            // ===== VPoint =====
            VizConsole.Log("Creating VPoint...");
            var point1 = new VPoint(0, 0);
            point1.Color = "Red";
            point1.Draw();

            var point2 = new VPoint(-200, 150);
            point2.Color = "Yellow";
            point2.Draw();

            // ===== VLine =====
            VizConsole.Log("Creating VLine...");
            var line1 = new VLine(0, 0, 100, 100);
            line1.Color = "Lime";
            line1.StrokeThickness = 2;
            line1.Draw();

            // Line from points
            var line2 = new VLine(new VPoint(-50, 0), new VPoint(-50, 100));
            line2.Color = "Cyan";
            line2.Draw();

            // ===== VCircle =====
            VizConsole.Log("Creating VCircle...");
            var circle1 = new VCircle(150, 100, 40);
            circle1.Color = "Magenta";
            circle1.FillColor = "Transparent";
            circle1.StrokeThickness = 3;
            circle1.Draw();

            // Filled circle
            var circle2 = new VCircle(150, 100, 20);
            circle2.Color = "White";
            circle2.FillColor = "Purple";
            circle2.Draw();

            // ===== VArc =====
            VizConsole.Log("Creating VArc...");
            var arc1 = new VArc(-150, 100, 50, 0, 180);
            arc1.Color = "Orange";
            arc1.StrokeThickness = 3;
            arc1.Draw();

            var arc2 = new VArc(-150, 100, 30, 45, 315);
            arc2.Color = "Gold";
            arc2.Draw();

            // ===== VRectangle =====
            VizConsole.Log("Creating VRectangle...");
            var rect1 = new VRectangle(-100, -50, 80, 40);
            rect1.Color = "DeepSkyBlue";
            rect1.FillColor = "DarkSlateBlue";
            rect1.Draw();

            var rect2 = new VRectangle(-90, -40, 60, 20);
            rect2.Color = "White";
            rect2.FillColor = "Transparent";
            rect2.StrokeThickness = 1;
            rect2.Draw();

            // ===== VEllipse =====
            VizConsole.Log("Creating VEllipse...");
            var ellipse1 = new VEllipse(50, -80, 60, 30);
            ellipse1.Color = "HotPink";
            ellipse1.FillColor = "Transparent";
            ellipse1.Draw();

            var ellipse2 = new VEllipse(50, -80, 40, 15);
            ellipse2.Color = "LightPink";
            ellipse2.FillColor = "MediumVioletRed";
            ellipse2.Draw();

            // ===== VPolygon =====
            VizConsole.Log("Creating VPolygon...");
            // Triangle
            var triangle = new VPolygon(
                new VPoint(-200, -100),
                new VPoint(-150, -50),
                new VPoint(-250, -50)
            );
            triangle.Color = "SpringGreen";
            triangle.FillColor = "DarkGreen";
            triangle.Draw();

            // Pentagon using List
            var pentagonPoints = new List<VPoint>();
            for (int i = 0; i < 5; i++)
            {
                double angle = i * 72 * Math.PI / 180 - Math.PI / 2;
                pentagonPoints.Add(new VPoint(
                    200 + Math.Cos(angle) * 40,
                    -80 + Math.Sin(angle) * 40
                ));
            }
            var pentagon = new VPolygon(pentagonPoints);
            pentagon.Color = "Coral";
            pentagon.FillColor = "DarkOrange";
            pentagon.Draw();

            // ===== VPolyline =====
            VizConsole.Log("Creating VPolyline...");
            var polyline = new VPolyline(
                new VPoint(-250, 50),
                new VPoint(-220, 80),
                new VPoint(-190, 50),
                new VPoint(-160, 80),
                new VPoint(-130, 50)
            );
            polyline.Color = "Aqua";
            polyline.StrokeThickness = 2;
            polyline.Draw();

            // ===== VBezier =====
            VizConsole.Log("Creating VBezier...");
            var bezier = new VBezier(
                new VPoint(-250, -150),
                new VPoint(-200, -200),
                new VPoint(-100, -100),
                new VPoint(-50, -150)
            );
            bezier.Color = "Violet";
            bezier.StrokeThickness = 3;
            bezier.Draw();

            // Show control points
            new VPoint(-250, -150) { Color = "Red" }.Draw();
            new VPoint(-200, -200) { Color = "Yellow" }.Draw();
            new VPoint(-100, -100) { Color = "Yellow" }.Draw();
            new VPoint(-50, -150) { Color = "Red" }.Draw();

            // ===== VSpline =====
            VizConsole.Log("Creating VSpline...");
            var splinePoints = new List<VPoint>
            {
                new VPoint(50, -150),
                new VPoint(100, -180),
                new VPoint(150, -130),
                new VPoint(200, -170),
                new VPoint(250, -140)
            };
            var spline = new VSpline(splinePoints);
            spline.Color = "Turquoise";
            spline.StrokeThickness = 3;
            spline.Draw();

            // Mark spline points
            foreach (var pt in splinePoints)
            {
                var marker = new VCircle(pt.X, pt.Y, 3);
                marker.Color = "White";
                marker.FillColor = "Turquoise";
                marker.Draw();
            }

            // ===== VArrow =====
            VizConsole.Log("Creating VArrow...");
            var arrow1 = new VArrow(-50, 150, 50, 150);
            arrow1.Color = "OrangeRed";
            arrow1.HeadLength = 15;
            arrow1.Draw();

            var arrow2 = new VArrow(100, 120, 100, 180);
            arrow2.Color = "LimeGreen";
            arrow2.StrokeThickness = 2;
            arrow2.Draw();

            // Diagonal arrow
            var arrow3 = new VArrow(130, 120, 180, 170);
            arrow3.Color = "DodgerBlue";
            arrow3.HeadLength = 12;
            arrow3.Draw();

            // ===== VDimension =====
            VizConsole.Log("Creating VDimension...");
            var dim1 = new VDimension(-100, 180, 0, 180);
            dim1.Offset = 20;
            dim1.Color = "Yellow";
            dim1.Draw();

            var dim2 = new VDimension(200, 50, 200, 150);
            dim2.Offset = 25;
            dim2.Color = "Cyan";
            dim2.Draw();

            // ===== VText =====
            VizConsole.Log("Creating VText...");
            var text1 = new VText(0, 200, "Viz2D Shape Demo");
            text1.Color = "White";
            text1.Height = 20;
            text1.Draw();

            var text2 = new VText(-250, 200, "Points & Lines");
            text2.Color = "Lime";
            text2.Height = 12;
            text2.Draw();

            var text3 = new VText(150, 200, "Curves & Arrows");
            text3.Color = "Magenta";
            text3.Height = 12;
            text3.Draw();

            // ===== VGroup =====
            VizConsole.Log("Creating VGroup...");
            // Create a grouped shape (face)
            var faceCircle = new VCircle(0, 50, 30);
            faceCircle.Color = "Yellow";
            faceCircle.FillColor = "Gold";

            var leftEye = new VCircle(-10, 60, 5);
            leftEye.Color = "Black";
            leftEye.FillColor = "Black";

            var rightEye = new VCircle(10, 60, 5);
            rightEye.Color = "Black";
            rightEye.FillColor = "Black";

            var smile = new VArc(0, 45, 15, 200, 340);
            smile.Color = "Black";
            smile.StrokeThickness = 2;

            var faceGroup = new VGroup(faceCircle, leftEye, rightEye, smile);
            faceGroup.Draw();

            // ===== Pattern: Star =====
            VizConsole.Log("Creating Star Pattern...");
            var starPoints = new List<VPoint>();
            int numPoints = 5;
            double outerRadius = 35;
            double innerRadius = 15;

            for (int i = 0; i < numPoints * 2; i++)
            {
                double radius = (i % 2 == 0) ? outerRadius : innerRadius;
                double angle = i * Math.PI / numPoints - Math.PI / 2;
                starPoints.Add(new VPoint(
                    -200 + Math.Cos(angle) * radius,
                    -150 + Math.Sin(angle) * radius
                ));
            }

            var star = new VPolygon(starPoints);
            star.Color = "Gold";
            star.FillColor = "DarkGoldenrod";
            star.Draw();

            // ===== Pattern: Radial =====
            VizConsole.Log("Creating Radial Pattern...");
            int count = 8;
            double radius = 25;
            for (int i = 0; i < count; i++)
            {
                double angle = i * 360.0 / count * Math.PI / 180;
                double x = 250 + Math.Cos(angle) * radius;
                double y = 100 + Math.Sin(angle) * radius;

                var dot = new VCircle(x, y, 5);
                dot.Color = "White";
                dot.FillColor = "DeepPink";
                dot.Draw();
            }

            // ===== Summary =====
            VizConsole.Log("-----------------------------------");
            VizConsole.Log("All shapes demonstrated:");
            VizConsole.Log("  - VPoint: Basic 2D point");
            VizConsole.Log("  - VLine: Line between two points");
            VizConsole.Log("  - VCircle: Circle with center and radius");
            VizConsole.Log("  - VArc: Arc with angles in degrees");
            VizConsole.Log("  - VRectangle: Rectangle with corner, width, height");
            VizConsole.Log("  - VEllipse: Ellipse with X/Y radii");
            VizConsole.Log("  - VPolygon: Closed polygon from points");
            VizConsole.Log("  - VPolyline: Open polyline from points");
            VizConsole.Log("  - VBezier: Cubic bezier curve");
            VizConsole.Log("  - VSpline: Smooth cardinal spline");
            VizConsole.Log("  - VArrow: Line with arrowhead");
            VizConsole.Log("  - VDimension: Dimension annotation");
            VizConsole.Log("  - VText: Text label");
            VizConsole.Log("  - VGroup: Group of shapes");
            VizConsole.Log("-----------------------------------");
            VizConsole.Log("Shape properties: Color, FillColor, StrokeThickness");
            VizConsole.Log("Call Draw() to render shapes on canvas");
        }
    }
}
